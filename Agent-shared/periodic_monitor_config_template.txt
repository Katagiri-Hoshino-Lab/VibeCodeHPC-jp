# periodic_monitor.sh 設定ファイルテンプレート
# このファイルをAgent-shared/periodic_monitor_config.txtとしてコピーして使用
# PM/SOLOエージェントが必要に応じて値を調整可能

# 上書き更新の頻度（秒）
# デフォルト: 30秒
# 推奨範囲: 10-300秒
UPDATE_INTERVAL_SEC=30

# マイルストーン確認間隔（分）
# デフォルト: 30分
# この間隔でマイルストーン（30,60,90,120,180分）をチェック
MILESTONE_INTERVAL_MIN=30

# 最大実行時間（分）
# デフォルト: 4320分（3日）
# プロジェクトの予想時間に応じて調整
MAX_RUNTIME_MIN=4320

# 予算集計の実行間隔（分）
# デフォルト: 3分
# 推奨範囲: 1-10分
BUDGET_INTERVAL_MIN=3

# 使用例:
# 短時間テスト用:
# UPDATE_INTERVAL_SEC=10
# MILESTONE_INTERVAL_MIN=5
# MAX_RUNTIME_MIN=60
#
# 長時間プロジェクト用:
# UPDATE_INTERVAL_SEC=60
# MILESTONE_INTERVAL_MIN=60
# MAX_RUNTIME_MIN=1440  # 24時間
